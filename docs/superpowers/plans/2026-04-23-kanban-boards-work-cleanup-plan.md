**Goal:** Generate Obsidian-friendly board views from issue Markdown, add the Work/Jira export adapter, and remove old engine-repo live-state assumptions from docs and scripts.

**Architecture:** Board generation reads canonical issues and writes generated Markdown under `boards/` while treating `spaces/` as legacy migration input. The Jira adapter exports canonical JSON to Jira payloads and optionally writes allowed metadata back to Markdown frontmatter. Cleanup updates docs, templates, and scripts to point at `KANBAN_HOME` instead of `~/Projects/kanban-task-engine/issues`.

**Tech Stack:** TypeScript, Vitest, Markdown, YAML frontmatter, existing adapter package patterns, pnpm workspaces.

```typescript
import { IssueStatus, ISSUE_STATUSES } from '@kanban-task-engine/schema';

export interface BoardIssue {
  id: string;
  title: string;
  status: IssueStatus;
  priority: string;
  project: string;
  path: string;
}

export interface RenderBoardOptions {
  boardTitle: string;
  generatedAt: string;
  issues: BoardIssue[];
}

export function renderBoardMarkdown(options: RenderBoardOptions): string {
  const lines: string[] = [
    '<!-- GENERATED BY kanban-task-engine. DO NOT EDIT DIRECTLY. -->',
    '---',
    'type: kanban-board',
    `title: ${options.boardTitle}`,
    `generatedAt: ${options.generatedAt}`,
    '---',
    '',
    `# ${options.boardTitle}`,
    '',
  ];

  for (const status of ISSUE_STATUSES) {
    lines.push(`## ${status}`, '');
    const issues = options.issues.filter(issue => issue.status === status);
    if (issues.length === 0) {
      lines.push('- No issues', '');
      continue;
    }
    for (const issue of issues) {
      lines.push(`- [${issue.title}](../${issue.path}) \`${issue.priority}\` <!-- ${issue.id} -->`);
    }
    lines.push('');
  }

  return `${lines.join('\n').trimEnd()}\n`;
}
```

```typescript
import { CanonicalTaskModel } from '@kanban-task-engine/core';

export interface JiraPayloadOptions {
  jiraProject: string;
}

export interface JiraIssuePayload {
  fields: {
    project: { key: string };
    summary: string;
    description: string;
    issuetype: { name: string };
    priority: { name: string };
    labels: string[];
  };
}

export function canonicalToJiraPayload(issue: CanonicalTaskModel, options: JiraPayloadOptions): JiraIssuePayload {
  return {
    fields: {
      project: { key: options.jiraProject },
      summary: issue.summary,
      description: issue.description_ref ?? '',
      issuetype: { name: issue.classification.issue_type },
      priority: { name: issue.classification.priority },
      labels: issue.classification.labels,
    },
  };
}
```

```typescript
import { JiraIssuePayload } from './jira-mapper';

export interface JiraAdapterConfig {
  baseUrl: string;
  token: string;
  dryRun?: boolean;
}

export interface JiraAdapterResult {
  dryRun: boolean;
  key?: string;
  payload: JiraIssuePayload;
}

export type JiraTransport = (url: string, init: RequestInit) => Promise<Response>;

export class JiraAdapter {
  constructor(
    private config: JiraAdapterConfig,
    private transport: JiraTransport = fetch
  ) {}

  async createIssue(payload: JiraIssuePayload): Promise<JiraAdapterResult> {
    if (this.config.dryRun) {
      return { dryRun: true, payload };
    }

    const response = await this.transport(`${this.config.baseUrl}/rest/api/3/issue`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.config.token}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(payload),
    });

    if (!response.ok) {
      throw new Error(`Jira create failed: ${response.status}`);
    }
    const data = await response.json() as { key?: string };
    return { dryRun: false, key: data.key, payload };
  }
}
```
